/* Provincia */
create table provincias(
id_provincia int,
constraint pk_id_provincia primary key (id_provincia),
provincia varchar (50),
)

/* Localidades */
create table localidades(
id_localidad int,
constraint pk_id_localidades primary key (id_localidad),
localidad varchar (40),
id_provincia int,
constraint fk_id_provincia foreign key (id_provincia) references
provincias (id_provincia)
)

/* Barrios */
create table barrios(
id_barrio int,
constraint pk_id_barrio primary key (id_barrio),
barrio varchar(40),
id_localidad int,
constraint fk_localidad foreign key (id_localidad) references
localidades (id_localidad)
)

create table tipos_examenes(
    id_tipo_examen int,
    descripcion varchar(30),

    constraint pk_tipos_examenes primary key (id_tipo_examen),
)

create table tecnicaturas(
    id_tecnicatura int identity (1,1),
    descripcion varchar(50),

    constraint pk_carreras primary key (id_tecnicatura),
)

create table materias(
    id_materia int identity (1,1),
    nombre_materia varchar(50),
    id_tecnicatura int,

    constraint pk_materias primary key (id_materia),
    constraint fk_materias_carreras foreign key (id_tecnicatura) REFERENCES tecnicaturas(id_tecnicatura)
)

create table estados_academicos(
    id_estado int,
    descripcion varchar(20),

    constraint pk_estados_academicos primary key (id_estado),
)

create table cursos(
    id_curso int,
    descripcion varchar(50),

    constraint pk_cursos primary key (id_curso),
)

create table tipos_dni(
    id_tipo_dni int,
    descripcion varchar(30),

    constraint pk_tipos_dni primary key (id_tipo_dni),
)

create table estados_civil(
    id_estado_civil int,
    descripcion varchar(30),

    constraint pk_estados_civil primary key (id_estado_civil),
)

create table situaciones_hab(
    id_situacion_habitacional int,
    descripcion varchar(30),

    constraint pk_situaciones_hab primary key (id_situacion_habitacional)
)

create table situaciones_lab(
    id_situacion_laboral int,
    descripcion varchar(30),

    constraint pl_situaciones_lab primary key (id_situacion_laboral),
)

create table Alumnos(
    id_alumno int identity (1,1),
    nombre varchar(30),
    apellido varchar(30),
    id_tipo_dni int,
    nro_dni varchar(10),
    id_tecnicatura int,
    fec_nac datetime,
    id_estado_civil int,
    id_situacion_laboral int,
    id_situacion_habitacional int,
    id_barrio int,
    direccion varchar(100),

    constraint pk_alumnos primary key (id_alumno),
    constraint fk_dni_alumnos foreign key (id_tipo_dni) REFERENCES tipos_dni(id_tipo_dni),
    constraint fk_carreras_alumnos foreign key (id_tecnicatura) References tecnicaturas(id_tecnicatura),
    constraint fk_estado_civil_alumnos foreign key (id_estado_civil) references estados_civil(id_estado_civil),
    constraint fk_situacion_lab_alumnos foreign key(id_situacion_laboral) references situaciones_lab(id_situacion_laboral),
    constraint fk_situacion_hab_alumnos foreign key(id_situacion_habitacional) references situaciones_hab(id_situacion_habitacional),
    constraint fk_barrios_alumnos foreign key (id_barrio) references barrios(id_barrio),
)

create table profesores(
    id_profesor int identity (1,1),
    nombre varchar(30),
    apellido varchar(30),
    id_tipo_dni int,
    nro_dni varchar(10),
    fec_nac datetime,
    id_estado_civil int,
    id_barrio int,
    direccion varchar(100),
    titulo_univ varchar(50),

    constraint pk_profesores primary key (id_profesor),
    constraint fk_dni_profesores foreign key (id_tipo_dni) REFERENCES tipos_dni(id_tipo_dni),
    constraint fk_estado_civil_profesores foreign key (id_estado_civil) references estados_civil(id_estado_civil),
    constraint fk_barrios_profesores foreign key (id_barrio) references barrios(id_barrio),
)

create table materias_profesores(
    id_materia_profesor int identity (1,1),
    año int,
    id_profesor int,
    id_materia int,

    constraint pk_materias_profesores primary key (id_materia_profesor),
    constraint fk_profesor_mp foreign key (id_profesor) references profesores(id_profesor),
    constraint fk_materias_mp foreign key (id_materia) references materias(id_materia),
)

create table alumnos_materias(
    id_alumno_materia int identity (1,1),
    año int,
    id_estado int,
    id_alumno int,
    id_materia int,
    id_curso int,

    constraint pk_alumnos_materias primary key (id_alumno_materia),
    constraint fk_estados_am foreign key (id_estado) references estados_academicos(id_estado),
    constraint fk_alumnos_am foreign key (id_alumno) references Alumnos(id_alumno),
    constraint fk_materias_am foreign key (id_materia) references materias(id_materia),
    constraint fk_cursos_am foreign key (id_curso) references cursos(id_curso),
)

create table examenes(
    id_examen int identity (1,1),
    id_tipo_examen int,
    id_materia int,
    fecha datetime,

    constraint pk_examenes primary key (id_examen),
    constraint fk_tipos_examen_detalle_examenes foreign key (id_tipo_examen) references tipos_examenes(id_tipo_examen),
    constraint fk_materias_detalle_examenes foreign key (id_materia) references materias(id_materia),

)

create table detalle_examenes(
    id_detalle int identity (1,1),
    id_examen int,
    id_alumno int,
    nota decimal,

    constraint pk_detalle_examenes primary key (id_detalle),
    constraint fk_examenes_detalles foreign key (id_examen) REFERENCES examenes(id_examen),
    constraint fk_alumnos_examenes foreign key (id_alumno) references Alumnos(id_alumno),

)
